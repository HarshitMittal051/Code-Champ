//Given a number N, print value of factorial N (N!)

#include <iostream>
using namespace std;
 
int main()
{
    int n;
    cin>>n;
    int a[200]={0};
    a[0]=1;
    long int len=1;
    if(n==0){
        cout<<"1";
        return 0;
    }
    
    for(int i=2;i<=n;i++){
     
	  for(int j=0;j<200;j++){  
        a[j]=a[j]*i;
      } 
	  
      for(int j=0;j<200;j++){
       if(a[j]>9){
            int x=a[j]/10;
            a[j]=a[j]%10;
            a[j+1] += x;
        }
      }
    }  
    //   if(a[len]!=0)
    //     len++;        
    len=199;
    while(a[len]==0)
    {
        len--;
    }
    
    for(int i=len;i>=0;i--)
        cout<<a[i];
    
    return 0;
}
