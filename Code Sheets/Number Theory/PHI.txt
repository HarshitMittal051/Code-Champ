//The Euler Totient (Ï†) of a positive integer N is - number of positive integers less than or equal to N that are coprime to N!
//Calculate the value of PHI for the given number!
#include <iostream>
using namespace std;

int gcd(int a,int b)
{
    if(b==0)
        return a;
        
    gcd(b,a%b);    
}

int main()
{
    //cout << "Hello World!" << endl;
    
    long int n,i;
    cin>>n;
    long int count=0;
    for(i=0;i<n;i++)
    {
        if(gcd(n,i)==1)
            count++;
    }
    
    cout<<count;
    
    return 0;
}
